version: "3.9"

services:
  mongo:
    image: mongo:6
    container_name: mongo
    restart: always
    ports:
      - "27017:27017"
    volumes:
      - mongo_data:/data/db

  backend:
    image: snehatrimukhe135/mean-backend:latest
    container_name: backend
    restart: always
    depends_on:
      - mongo
    environment:
      # Adjust DB name as per your code
      - MONGO_URI=mongodb://mongo:27017/mean-crud-db
      - PORT=3000
    # if you want logs
    # ports:
    #   - "3000:3000"

  frontend:
    image: snehatrimukhe135/mean-frontend:latest
    container_name: frontend
    restart: always
    depends_on:
      - backend
    # Angular is served by nginx inside this container on port 80
    # ports:
    #   - "4200:80"

  nginx:
    image: nginx:alpine
    container_name: reverse-proxy
    depends_on:
      - frontend
      - backend
    ports:
      - "80:80"      # Entire app available on port 80
    volumes:
      - ./nginx/default.conf:/etc/nginx/conf.d/default.conf:ro
    restart: always

volumes:
  mongo_data:
